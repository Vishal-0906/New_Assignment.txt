Question 1
Use fetch() to get a list of posts from https://jsonplaceholder.typicode.com/posts and display the first 5 post titles in the browser console.
Solution
HTML
<!DOCTYPE html>
<html>
<head>
    <title>Question 1</title>
</head>
<body>

<script>
fetch("https://jsonplaceholder.typicode.com/posts")
.then(response => response.json())
.then(data => {
    data.slice(0,5).forEach(post => {
        console.log(post.title);
    });
})
.catch(error => console.log(error));
</script>

</body>
</html>

Question 5
Build a function fetchTopMovies() that uses fetch() to get movie data from https://api.tvmaze.com/shows, parses the JSON response, and logs the titles of the first 3 items.
Solution
async function fetchTopMovies(){

try{

let response=await fetch("https://api.tvmaze.com/shows");

let data=await response.json();

data.slice(0,3).forEach(movie=>{
console.log(movie.name);
});

}
catch(error){
console.log(error);
}

}

fetchTopMovies();

Question 9
Chain fetch() promises to fetch user data from https://jsonplaceholder.typicode.com/users, then display all user names in a single line separated by commas.
Solution
fetch("https://jsonplaceholder.typicode.com/users")
.then(response=>response.json())
.then(users=>{

let names=users.map(user=>user.name);

console.log(names.join(", "));

})
.catch(error=>console.log(error));

Question 10
Chain two fetch() calls: first fetch users from /users, then fetch posts for userId=2, and display the user's name and the titles of their posts.
Solution
fetch("https://jsonplaceholder.typicode.com/users")
.then(response=>response.json())
.then(users=>{

let user=users.find(u=>u.id===2);

console.log("User:",user.name);

return fetch("https://jsonplaceholder.typicode.com/posts?userId=2");

})
.then(response=>response.json())
.then(posts=>{

posts.forEach(post=>{
console.log(post.title);
});

})
.catch(error=>console.log(error));

Question 15
Fetch a list of albums from https://jsonplaceholder.typicode.com/albums, filter the albums where userId = 2, and display only the album titles using fetch(), .then(), and array methods (no async/await).
Solution
fetch("https://jsonplaceholder.typicode.com/albums")
.then(response=>response.json())
.then(albums=>{

let result=albums.filter(album=>album.userId===2);

result.forEach(album=>{
console.log(album.title);
});

})
.catch(error=>console.log(error));
