Q10. Create two variables: let user1 = 'Priya' and let user2 = user1. Change user1 to 'Riya' and print both variables. Then, create two arrays: let favApps1 = ['Instagram', 'Spotify'] and let favApps2 = favApps1. Push 'Zomato' to favApps1 and print both arrays. Observe and comment on the difference in behavior between primitive and reference types.

Answer:
`javascript
// Primitive type example
let user1 = "Priya";
let user2 = user1;
user1 = "Riya";

console.log("user1:", user1); // Riya
console.log("user2:", user2); // Priya

// Reference type example
let favApps1 = ["Instagram", "Spotify"];
let favApps2 = favApps1;

favApps1.push("Zomato");

console.log("favApps1:", favApps1); // ["Instagram", "Spotify", "Zomato"]
console.log("favApps2:", favApps2); // ["Instagram", "Spotify", "Zomato"]

12. Show the difference between primitive and reference types by creating a variable favFood = 'pizza' and another variable favFoodCopy = favFood. Change favFood to 'burger' and log both variables. Then, create an object zomatoUser with a property city, assign it to another variable, change the city in one, and log both objects.

Answer:
`javascript
// Primitive type example
let favFood = "pizza";
let favFoodCopy = favFood;
favFood = "burger";

console.log("favFood:", favFood);       // burger
console.log("favFoodCopy:", favFoodCopy); // pizza

// Reference type example
let zomatoUser = { city: "Delhi" };
let zomatoUserCopy = zomatoUser;

zomatoUser.city = "Mumbai";

console.log("zomatoUser:", zomatoUser);       // { city: "Mumbai" }
console.log("zomatoUserCopy:", zomatoUserCopy); // { city: "Mumbai" }

Q13. Write a function showProfile that takes a name, followers, and isPremium (boolean) as arguments. Inside the function, use dynamic typing to assign a string to followers, then a number, and finally print the type of followers each time using typeof.

Answer:
`javascript
function showProfile(name, followers, isPremium) {
  followers = "One Thousand"; 
  console.log("Followers:", followers, "Type:", typeof followers);

  followers = 1000; 
  console.log("Followers:", followers, "Type:", typeof followers);

  console.log("Name:", name, "isPremium:", isPremium);
}

showProfile("Vishal", 1500, true);
`
