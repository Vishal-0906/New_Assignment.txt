Q4
Question: Add a method displayPlaylist() to the Playlist class that logs the playlist name and all song titles in a formatted string.  

Answer:
`javascript
class Playlist {
  constructor(name, songs) {
    this.name = name;
    this.songs = songs;
  }

  displayPlaylist() {
    console.log(Playlist: ${this.name});
    console.log("Songs:");
    this.songs.forEach(song => console.log("- " + song));
  }
}

const chillVibes = new Playlist("Chill Vibes", ["Song A", "Song B", "Song C"]);
chillVibes.displayPlaylist();
`

Q7
Question: Create a class called FlipkartProduct with properties: title, price, and rating. Write a method getDiscountedPrice(discountPercent) that returns the price after applying the given discount percentage.  

Answer:
`javascript
class FlipkartProduct {
  constructor(title, price, rating) {
    this.title = title;
    this.price = price;
    this.rating = rating;
  }

  getDiscountedPrice(discountPercent) {
    return this.price - (this.price * discountPercent / 100);
  }
}

const product = new FlipkartProduct("Shoes", 2000, 4.3);
console.log("Discounted Price:", product.getDiscountedPrice(10));
`

Q12
Question: Extend the Playlist class by adding a method addSong(songName) that adds a new song to the songs array. Demonstrate by adding two songs and displaying the updated playlist.  

Answer:
`javascript
class Playlist {
  constructor(name, songs) {
    this.name = name;
    this.songs = songs;
  }

  addSong(songName) {
    this.songs.push(songName);
  }

  displayPlaylist() {
    console.log(Playlist: ${this.name});
    this.songs.forEach(song => console.log("- " + song));
  }
}

const myPlaylist = new Playlist("Chill Vibes", ["Song A", "Song B"]);
myPlaylist.addSong("Song C");
myPlaylist.addSong("Song D");
myPlaylist.displayPlaylist();
`
