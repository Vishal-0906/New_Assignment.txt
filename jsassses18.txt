Q6
Question: Write a function addToCart(productName, quantity) that logs a success message using console.log(), a warning if quantity is zero using console.warn(), and an error if quantity is negative using console.error(). Call the function with all three scenarios.  

Answer:
`javascript
function addToCart(productName, quantity) {
  if (quantity > 0) {
    console.log(✅ Added ${quantity} ${productName}(s) to cart.);
  } else if (quantity === 0) {
    console.warn(⚠️ Quantity for ${productName} is zero.);
  } else {
    console.error(❌ Invalid quantity for ${productName}.);
  }
}

addToCart("Shoes", 2);   // Success
addToCart("Watch", 0);   // Warning
addToCart("Bag", -1);    // Error
`
