Q4. Build a small form with an input box and a button labeled 'Add Song'. When the button is clicked, use querySelector to get the input value and dynamically add a new <li> element with the song name to a <ul id="playlist"> on the page.

`html
<!DOCTYPE html>
<html>
<head>
  <title>Playlist</title>
</head>
<body>
  <input type="text" id="song-input" placeholder="Enter song name">
  <button id="add-song">Add Song</button>
  <ul id="playlist"></ul>

  <script>
    document.querySelector("#add-song").addEventListener("click", function() {
      const songName = document.querySelector("#song-input").value;
      if (songName.trim() !== "") {
        const li = document.createElement("li");
        li.textContent = songName;
        document.querySelector("#playlist").appendChild(li);
      }
    });
  </script>
</body>
</html>
`

Q9. Create a 'Dark Mode' toggle button that, when clicked, uses getElementById to change the background color of the page and updates the button text to 'Light Mode'.

`html
<!DOCTYPE html>
<html>
<head>
  <title>Dark Mode Toggle</title>
</head>
<body>
  <button id="toggle-mode">Dark Mode</button>

  <script>
    const button = document.getElementById("toggle-mode");
    button.addEventListener("click", function() {
      if (document.body.style.backgroundColor === "black") {
        document.body.style.backgroundColor = "white";
        button.textContent = "Dark Mode";
      } else {
        document.body.style.backgroundColor = "black";
        button.textContent = "Light Mode";
      }
    });
  </script>
</body>
</html>
`

Q10. Create a function that updates the background color of a <div id="moodbox"></div> in real-time as the user types a color name (like 'pink' or 'yellow') into an input box.

`html
<!DOCTYPE html>
<html>

  <script>
    const input = document.getElementById("color-input");
    const moodbox = document.getElementById("moodbox");

    input.addEventListener("input", function() {
      moodbox.style.backgroundColor = input.value;
    });
  </script>
</body>
</html>

Q12. Build a mini 'Now Playing' display like Spotify: show a song title in a <span id="song-title"></span> and a button 'Change Song'; when the button is clicked, update the innerHTML of the span to a new song name from a predefined array.


<!DOCTYPE html>
<html>
<head>
  <title>Now Playing</title>
</head>
<body>
  <h3>🎵 Now Playing:</h3>
  <span id="song-title">Song A</span>
  <br><br>
  <button id="change-song">Change Song</button>

  <script>
    const songs = ["Song A", "Song B", "Song C", "Song D"];
    let currentIndex = 0;

    const songTitle = document.getElementById("song-title");
    const changeButton = document.getElementById("change-song");

    changeButton.addEventListener("click", function() {
      // Move to the next song in the array
      currentIndex = (currentIndex + 1) % songs.length;
      songTitle.innerHTML = songs[currentIndex];
    });
  </script>
</body>
</html>
