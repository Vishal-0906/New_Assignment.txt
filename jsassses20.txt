Q4
Question: Create a login form with email and password fields and add basic UI validation: show an error message if either field is left empty when the user clicks the Login button.  

Answer:
`html
<form id="loginForm">
  <input type="email" id="email" placeholder="Email">
  <span id="emailError"></span><br>
  <input type="password" id="password" placeholder="Password">
  <span id="passwordError"></span><br>
  <button type="submit">Login</button>
</form>

<script>
  document.getElementById("loginForm").addEventListener("submit", function(e) {
    e.preventDefault();
    let email = document.getElementById("email").value;
    let password = document.getElementById("password").value;
    let emailError = document.getElementById("emailError");
    let passwordError = document.getElementById("passwordError");

    emailError.textContent = "";
    passwordError.textContent = "";

    if (!email) emailError.textContent = "Email is required!";
    if (!password) passwordError.textContent = "Password is required!";
  });
</script>
`

Q9
Question: Find and fix a bug in this code: the function addFollower(name) is supposed to add a name to the followers array, but sometimes null or empty names get added. Update the function so only valid, non-empty names are added.  

Answer:
`javascript
const followers = [];

function addFollower(name) {
  if (name && name.trim() !== "") {
    followers.push(name);
    console.log("Follower added:", name);
  } else {
    console.warn("Invalid follower name, not added.");
  }
}

addFollower("Amit");   // Valid
addFollower("");       // Invalid
addFollower(null);     // Invalid
`
