Question 4
Simulate a Zomato order status updater using callbacks. Write a function placeOrder(item, callback) that takes a food item and a callback, waits 1 second using setTimeout(), then calls the callback with the message "Your [item] is being prepared".
Solution
function placeOrder(item, callback)
{
    setTimeout(() => {
        callback("Your " + item + " is being prepared.");
    },1000);
}

placeOrder("Pizza", function(message){
    console.log(message);
});
Output
Your Pizza is being prepared.

Question 7
Build a Promise called getFollowersCount that resolves after 1 second with a random number between 1000 and 5000. Then use .then() to display "You have X new followers".
Solution
let getFollowersCount = new Promise((resolve, reject)=>{

setTimeout(()=>{

let followers = Math.floor(Math.random()*4001)+1000;

resolve(followers);

},1000);

});

getFollowersCount.then((count)=>{
console.log("You have " + count + " new followers.");
});

Question 9
Build a function getOrderStatus(orderId) that returns a Promise. Simulate a 1.5 second delay using setTimeout(), then resolve with "Order delivered" if the order ID is even, otherwise reject with "Order not found". Use .then() and .catch() to handle both cases.
Solution
function getOrderStatus(orderId)
{
return new Promise((resolve,reject)=>{

setTimeout(()=>{

if(orderId%2==0)
{
resolve("Order Delivered");
}
else
{
reject("Order Not Found");
}

},1500);

});
}

getOrderStatus(10)
.then((msg)=>{
console.log(msg);
})
.catch((err)=>{
console.log(err);
});
Question 10
Build a Flipkart-style product search using both callbacks and Promises. Given an array of products, simulate a 1-second delay before returning all products whose names include the search term.
Solution
let products = [
"Mobile",
"Laptop",
"Mouse",
"Keyboard",
"Smart Watch"
];

function searchProduct(search,callback)
{
setTimeout(()=>{

let result=products.filter(item=>item.toLowerCase().includes(search.toLowerCase()));

callback(result);

},1000);
}

searchProduct("mo",(result)=>{
console.log(result);
});
Output
["Mobile","Mouse"]

Question 12
Refactor the callback-based code below to use Promises instead.
Callback Code
setTimeout(function () {
    console.log("Payment successful");
}, 1000);
Solution
function paymentSuccess()
{
return new Promise((resolve,reject)=>{

setTimeout(()=>{

resolve("Payment Successful");

},1000);

});
}

paymentSuccess()
.then((message)=>{
console.log(message);
})
.catch((error)=>{
console.log(error);
});
