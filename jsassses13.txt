**Create a function validateSpotifyURL(url) that takes a string and checks if it starts with https://open.spotify.com/. If not, throw a custom error with the message "Invalid Spotify URL".
Solution
function validateSpotifyURL(url) {
    if (!url.startsWith("https://open.spotify.com/")) {
        throw new Error("Invalid Spotify URL");
    }

    return "Valid Spotify URL";
}

try {
    console.log(validateSpotifyURL("https://open.spotify.com/track/123"));
}
catch (error) {
    console.log(error.message);
}

Question 3
Create a function validateUsername(username) that checks whether the username contains only letters and numbers. If not, throw an error with the message "Invalid Username" and handle it using try...catch.
Solution
function validateUsername(username) {

    let pattern = /^[a-zA-Z0-9]+$/;

    if (!pattern.test(username)) {
        throw new Error("Invalid Username");
    }

    return "Valid Username";
}

try {
    console.log(validateUsername("Vishal123"));
}
catch (error) {
    console.log(error.message);
}

Write a JavaScript program that divides two numbers. If the denominator is 0, throw an error with the message "Division by zero is not allowed" and handle it using try...catch.
Solution
function divide(a, b) {
    try {
        if (b === 0) {
            throw new Error("Division by zero is not allowed");
        }

        console.log("Result:", a / b);
    }
    catch (error) {
        console.log(error.message);
    }
}

divide(20, 5);
divide(20, 0);


Question 8
Write a JavaScript program to access an array element using an index. If the index is out of range, throw an error with the message "Index Out of Bounds" and handle it using try...catch.
Solution
function getElement(arr, index) {
    try {
        if (index < 0 || index >= arr.length) {
            throw new Error("Index Out of Bounds");
        }

        console.log(arr[index]);
    }
    catch (error) {
        console.log(error.message);
    }
}

let numbers = [10, 20, 30, 40, 50];

getElement(numbers, 2);
getElement(numbers, 10);

You are given this buggy code: function formatFollowers (count) { try { if (count < 0) throw "Negative count"; return count + " followers"; } catch(e) { return e.message; } } It should return the error message if count is negative, but it returns undefined. Fix the code so it properly throws and catches an Error object.
Note: Use Error objects instead of strings for throwing and catching errors.

Answer:
function formatFollowers(count) {
    try {
        if (count < 0) {
            throw new Error("Negative count");
        }
        return count + " followers";
    } catch (e) {
        return e.message;
    }
}
