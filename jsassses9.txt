Question 6
Build a function saveRecentSearch(query) that takes a search term (like "Biryani" or "Sneakers"), adds it to an array, converts the array to JSON, and saves it in localStorage under the key recentSearches.
Solution
function saveRecentSearch(query) {

    let searches =
    JSON.parse(localStorage.getItem("recentSearches")) || [];

    searches.push(query);

    localStorage.setItem(
        "recentSearches",
        JSON.stringify(searches)
    );
}

saveRecentSearch("Biryani");
saveRecentSearch("Sneakers");
Question 7
Build a simple web page where users can add their top 3 favorite Zomato restaurants to a list and save it to localStorage using localStorage.setItem(). When the page reloads, the list should automatically load from localStorage and display.
HTML
HTML
<input type="text" id="restaurant">

<button onclick="addRestaurant()">
Add Restaurant
</button>

<ul id="list"></ul>
JavaScript
let restaurants =
JSON.parse(localStorage.getItem("restaurants")) || [];

display();

function addRestaurant() {

    let name =
    document.getElementById("restaurant").value;

    if(restaurants.length < 3){

        restaurants.push(name);

        localStorage.setItem(
            "restaurants",
            JSON.stringify(restaurants)
        );

        display();
    }
}

function display(){

    let list = document.getElementById("list");

    list.innerHTML = "";

    restaurants.forEach(function(item){

        list.innerHTML += `<li>${item}</li>`;

    });

}

Question 9
Write a function getRecentSearches() that reads the recentSearches item from localStorage, parses it from JSON, and returns the array. If nothing is saved yet, return an empty array.
Solution
function getRecentSearches(){

    let data =
    localStorage.getItem("recentSearches");

    if(data == null){

        return [];

    }

    return JSON.parse(data);

}

console.log(getRecentSearches());
