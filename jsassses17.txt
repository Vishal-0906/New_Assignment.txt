Q1
Question: Create a list of song cards (each with a title and an 'X' icon) using plain JavaScript and HTML. When you click the 'X' on any card, remove only that card from the DOM using event delegation (attach the click listener to the parent container, not each card).  

Answer:
`html
<div id="song-list">
  <div class="song-card">Song A <span class="remove">X</span></div>
  <div class="song-card">Song B <span class="remove">X</span></div>
  <div class="song-card">Song C <span class="remove">X</span></div>
</div>

<script>
  const songList = document.getElementById("song-list");

  songList.addEventListener("click", function(event) {
    if (event.target.classList.contains("remove")) {
      event.target.parentElement.remove();
    }
  });
</script>
`

Q4
Question: Build a Flipkart-style wishlist where each product card has a 'Remove' button. Add a new product card dynamically when the user clicks 'Add Product'. Ensure that the 'Remove' button works for both existing and newly added cards without attaching new listeners to each card.  

Answer:
`html
<div id="wishlist">
  <div class="product">Shoes <button class="remove">Remove</button></div>
</div>
<button id="add-product">Add Product</button>

<script>
  const wishlist = document.getElementById("wishlist");
  const addBtn = document.getElementById("add-product");

  wishlist.addEventListener("click", function(e) {
    if (e.target.classList.contains("remove")) {
      e.target.parentElement.remove();
    }
  });

  addBtn.addEventListener("click", function() {
    const newProduct = document.createElement("div");
    newProduct.className = "product";
    newProduct.innerHTML = "New Product <button class='remove'>Remove</button>";
    wishlist.appendChild(newProduct);
  });
</script>
`

Q10
Question: Simulate a notification feed like YouTube, where each notification card has a 'Dismiss' button. When you click 'Dismiss', fade out and remove the card. Make sure the event delegation works for dynamically loaded notifications as well as those present at page load.  

Answer:
`html
<div id="notifications">
  <div class="notification">New Video <button class="dismiss">Dismiss</button></div>
</div>
<button id="add-notification">Add Notification</button>

<script>
  const feed = document.getElementById("notifications");
  const addBtn = document.getElementById("add-notification");

  feed.addEventListener("click", function(e) {
    if (e.target.classList.contains("dismiss")) {
      const card = e.target.parentElement;
      card.style.transition = "opacity 0.5s";
      card.style.opacity = "0";
      setTimeout(() => card.remove(), 500);
    }
  });

  addBtn.addEventListener("click", function() {
    const newCard = document.createElement("div");
    newCard.className = "notification";
    newCard.innerHTML = "Random Alert <button class='dismiss'>Dismiss</button>";
    feed.appendChild(newCard);
  });
</script>
`

Q12
Question: Given the following code snippet, fix the bug so that clicking the 'X' icon on any card removes only that card, even if cards are added dynamically:  

`javascript
document.querySelectorAll('.remove-btn').forEach(btn => {
  btn.addEventListener('click', function() {
    this.parentElement.remove();
  });
});
`

Hint: Use event delegation by adding a single event listener to the parent container instead of looping through all buttons.  

Answer:
`html
<div id="card-container">
  <div class="card">Card 1 <button class="remove-btn">X</button></div>
  <div class="card">Card 2 <button class="remove-btn">X</button></div>
</div>
<button id="add-card">Add Card</button>

<script>
  const container = document.getElementById("card-container");
  const addBtn = document.getElementById("add-card");

  // Event delegation: single listener on parent
  container.addEventListener("click", function(e) {
    if (e.target.classList.contains("remove-btn")) {
      e.target.parentElement.remove();
    }
  });

  // Dynamically add new cards
  addBtn.addEventListener("click", function() {
    const newCard = document.createElement("div");
    newCard.className = "card";
    newCard.innerHTML = "New Card <button class='remove-btn'>X</button>";
    container.appendChild(newCard);
  });
</script>
