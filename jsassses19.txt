Q1
Question: Create an HTML file with a simple form that lets a user add a new playlist (name, description) like Spotify's 'Create Playlist' feature, and use JavaScript to store the playlist data in an array.  

Answer:
`html
<!DOCTYPE html>
<html>
<head>
  <title>Create Playlist</title>
</head>
<body>
  <h2>Create Playlist</h2>
  <form id="playlistForm">
    <input type="text" id="name" placeholder="Playlist Name" required>
    <input type="text" id="description" placeholder="Description" required>
    <button type="submit">Add Playlist</button>
  </form>

  <script>
    const playlists = [];

    document.getElementById("playlistForm").addEventListener("submit", function(e) {
      e.preventDefault();
      const name = document.getElementById("name").value;
      const description = document.getElementById("description").value;

      playlists.push({ name, description });
      console.log(playlists);
    });
  </script>
</body>
</html>
`

Q7
Question: Add 'Edit' and 'Delete' buttons next to each playlist in the displayed list, and implement the JavaScript functions so users can update or remove a playlist from the array and see the changes instantly.  

Answer:
`javascript
function renderPlaylists() {
  const display = document.getElementById("playlistDisplay");
  display.innerHTML = "";
  playlists.forEach((p, index) => {
    const div = document.createElement("div");
    div.innerHTML = `${p.name} - ${p.description} 
      <button onclick="editPlaylist(${index})">Edit</button>
      <button onclick="deletePlaylist(${index})">Delete</button>`;
    display.appendChild(div);
  });
}

function editPlaylist(index) {
  const playlist = playlists[index];
  document.getElementById("name").value = playlist.name;
  document.getElementById("description").value = playlist.description;
  playlists.splice(index, 1); // remove old entry, will be replaced on submit
}

function deletePlaylist(index) {
  playlists.splice(index, 1);
  renderPlaylists();
}
`

Q13
Question: Add an Edit feature: when the Edit button is clicked for a restaurant, pre-fill the form with that restaurant's details, allow the user to update them, and save the changes back to the array and UI.  

Answer:
`html
<form id="restaurantForm">
  <input type="text" id="resName" placeholder="Restaurant Name" required>
  <input type="text" id="cuisine" placeholder="Cuisine" required>
  <input type="number" id="rating" placeholder="Rating" required>
  <button type="submit">Save Restaurant</button>
</form>

<div id="restaurantList"></div>

<script>
  const restaurants = [];
  let editIndex = null;

  document.getElementById("restaurantForm").addEventListener("submit", function(e) {
    e.preventDefault();
    const resName = document.getElementById("resName").value;
    const cuisine = document.getElementById("cuisine").value;
    const rating = document.getElementById("rating").value;

    if (editIndex !== null) {
      // update existing restaurant
      restaurants[editIndex] = { resName, cuisine, rating };
      editIndex = null;
    } else {
      // add new restaurant
      restaurants.push({ resName, cuisine, rating });
    }

    renderRestaurants();
    e.target.reset();
  });

  function renderRestaurants() {
    const display = document.getElementById("restaurantList");
    display.innerHTML = "";
    restaurants.forEach((r, index) => {
      const div = document.createElement("div");
      div.innerHTML = `${r.resName} - ${r.cuisine} - ${r.rating}
        <button onclick="editRestaurant(${index})">Edit</button>
        <button onclick="deleteRestaurant(${index})">Delete</button>`;
      display.appendChild(div);
    });
  }

  function editRestaurant(index) {
    const res = restaurants[index];
    document.getElementById("resName").value = res.resName;
    document.getElementById("cuisine").value = res.cuisine;
    document.getElementById("rating").value = res.rating;
    editIndex = index; // mark index for update
  }

  function deleteRestaurant(index) {
    restaurants.splice(index, 1);
    renderRestaurants();
  }
</script>
