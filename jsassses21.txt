Q1

Question: Create a JavaScript function addPlaylistLink(linkObj) that adds a new music playlist link (with name and URL) to an array and saves the updated array to LocalStorage under the key 'myPlaylists'.

Answer:

function addPlaylistLink(linkObj) {
  let playlists = JSON.parse(localStorage.getItem("myPlaylists")) || [];
  playlists.push(linkObj);
  localStorage.setItem("myPlaylists", JSON.stringify(playlists));
}

Question: Build a function editPlaylistLink(index, newLinkObj) that updates the name and URL of a playlist link at a given index in your 'myPlaylists' array, then persists the changes to LocalStorage.

}

Q7

Question: Implement a toggleTheme() function that switches between light and dark mode by adding or removing a 'dark-mode' class on the body element, and persists the current theme in LocalStorage.

Answer:

function toggleTheme() {
  document.body.classList.toggle("dark-mode");
  let theme = document.body.classList.contains("dark-mode") ? "dark" : "light";
  localStorage.setItem("theme", theme);
}

Q8

Question: Implement a toggleTheme() function that switches between light and dark mode by adding or removing a 'dark-mode' class on the body element, and persists the current theme in LocalStorage.

Answer:

function toggleTheme() {
  if (document.body.classList.contains("dark-mode")) {
    document.body.classList.remove("dark-mode");
    localStorage.setItem("theme", "light");
  } else {
    document.body.classList.add("dark-mode");
    localStorage.setItem("theme", "dark");
  }
}

function manageLink(linkObj, mode, index = null) {
  let playlists = JSON.parse(localStorage.getItem("myPlaylists")) || [];

  if (mode === "add") {
    playlists.push(linkObj);
  } else if (mode === "edit" && index !== null) {

  localStorage.setItem("myPlaylists", JSON.stringify(playlists));
}

Q12

Question: Implement a toggleTheme() function that switches between light and dark mode for your playlist links page, saving the selected mode in LocalStorage and applying it on next visit.

Answer:

function toggleTheme() {
  document.body.classList.toggle("dark-mode");
  let theme = document.body.classList.contains("dark-mode") ? "dark" : "light";
  localStorage.setItem("theme", theme);
}

// Apply saved theme on page load
window.onload = function() {
  let savedTheme = localStorage.getItem("theme");
  if (savedTheme === "dark") {
    document.body.classList.add("dark-mode");
  }
};

Question: Use ChatGPT or Copilot to generate a JavaScript code snippet that displays an error message below a form input if the user enters an invalid Instagram profile URL, and explain in 2 lines how you would integrate this snippet into your existing code.

Answer:

function validateInstagramURL(url) {
  const errorMsg = document.getElementById("instaError");
  if (!url.startsWith("https://www.instagram.com/")) {
    errorMsg.textContent = "Invalid Instagram profile URL!";
  } else {
    errorMsg.textContent = "";
  }
}

Explanation: Call validateInstagramURL() on form submission or input blur event. Add a <span id="instaError"></span> below the input field to show the error.

Q15

Question: Build a validateLink(url) function that checks if a playlist URL starts with 'https://' and contains 'spotify.com' or 'youtube.com'. Show an error message if validation fails.

Answer:

function validateLink(url) {
  const error = document.getElementById("linkError");
  if (url.startsWith("https://") && (url.includes("spotify.com") || url.includes("youtube.com"))) {
    error.textContent = "";
    return true;
  } else {
    error.textContent = "Invalid playlist URL. Must be Spotify or YouTube.";
    return false;
  }
}

Usage:

<input type="text" id="playlistURL" placeholder="Enter playlist URL">
<span id="linkError"></span>
